SELECT  I.id, I.data, I.modalita_pagamento, I.descrizione, I.importo, M.nome, M.indirizzo, M.citta, M.p_iva, M.cap , M.provincia, I.created_at, A.nome, A.cognome, B.cod_fiscale, B.p_iva
FROM transactions as I, asds as M, users as A, members as B
WHERE I.id=M.id AND I.id=A.id AND A.id=B.id AND tipo_transazione='entrata'
